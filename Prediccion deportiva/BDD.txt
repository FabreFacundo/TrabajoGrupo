CREATE DATABASE Apuestas;
CREATE TABLE Apuestas.Personas
( 
idPersona INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
Nombre VARCHAR(45) NOT NULL
);
CREATE TABLE Apuestas.Pronosticos
( 
idPronostico INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
nroRonda INT NOT NULL DEFAULT '0',
nroPartido INT NOT NULL DEFAULT '0',
Equipo VARCHAR(45) NOT NULL,
Resultado VARCHAR(45) NOT NULL,
idPersona INT NOT NULL,
FOREIGN KEY (idPersona) REFERENCES Apuestas.Personas(idPersona) 
ON UPDATE CASCADE ON DELETE CASCADE
)
